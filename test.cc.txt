#include <iostream>
#include <vector>
using namespace std;
int main() {
    long long start, end; 
    cout << "Введите начало диапазона: ";
    if (!(cin >> start)) {
        cout << "Ошибка: введите целое число." << endl;
        return 1;
    }
    cout << "Введите конец диапазона: ";
    if (!(cin >> end)) {
        cout << "Ошибка: введите целое число." << endl;
        return 1;
    }
    if (start > end) {
        cout << "Ошибка: начало диапазона не может быть больше конца" << endl;
        return 1;
    }
    vector<long long> fibonacciNumbers;
    long long a = 0, b = 1;
   
    if (a >= start && a <= end) {
        fibonacciNumbers.push_back(a);
    }
   
    while (b <= end) { 
        if (b >= start) {
            fibonacciNumbers.push_back(b);
        }
        long long next = a + b;
        a = b;
        b = next;
    }
    
    if (fibonacciNumbers.empty()) {
        cout << "В диапазоне [" << start << ", " << end << "] нет чисел Фибоначчи." << endl;
    } else {
        cout << "Числа Фибоначчи в диапазоне [" << start << ", " << end << "]: " << endl;
        for (long long num : fibonacciNumbers) {
            cout << num << " ";
        }
        cout << endl;
    }
    return 0;
}
